CREATE TABLE `fase_realizada` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `piezas` varchar(45) COLLATE utf8_spanish_ci DEFAULT NULL,
  `precio` float DEFAULT NULL,
  `color` varchar(50) COLLATE utf8_spanish_ci DEFAULT NULL,
  `costo_comision` float DEFAULT NULL,
  `inicio` datetime DEFAULT NULL,
  `fin` datetime DEFAULT NULL,
  `detalle` varchar(45) COLLATE utf8_spanish_ci DEFAULT NULL,
  `codigo_empleado_fr` varchar(10) COLLATE utf8_spanish_ci NOT NULL,
  `id_prueba_realizada_fr` int(11) NOT NULL,
  `codigo_fase_fr` int(11) NOT NULL,
  PRIMARY KEY (`id`),
  KEY `fk_fase_realizada_empleado1_idx` (`codigo_empleado_fr`),
  KEY `fk_fase_realizada_prueba_realizada1_idx` (`id_prueba_realizada_fr`),
  KEY `fk_fase_realizada_fase1_idx` (`codigo_fase_fr`),
  CONSTRAINT `fk_fase_realizada_empleado1` FOREIGN KEY (`codigo_empleado_fr`) REFERENCES `empleado` (`codigo`) ON DELETE NO ACTION ON UPDATE NO ACTION,
  CONSTRAINT `fk_fase_realizada_fase1` FOREIGN KEY (`codigo_fase_fr`) REFERENCES `fase` (`codigo`) ON DELETE NO ACTION ON UPDATE NO ACTION,
  CONSTRAINT `fk_fase_realizada_prueba_realizada1` FOREIGN KEY (`id_prueba_realizada_fr`) REFERENCES `prueba_realizada` (`id`) ON DELETE NO ACTION ON UPDATE NO ACTION
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_spanish_ci
