CREATE TABLE `producto_solicitado` (
  `id` varchar(12) COLLATE utf8_spanish_ci NOT NULL,
  `posicion` tinyint(1) unsigned NOT NULL,
  `fecha_registro` date DEFAULT NULL,
  `fecha_entrega` date DEFAULT NULL,
  `cant_pres` tinyint(1) unsigned DEFAULT NULL,
  `precio` float NOT NULL,
  `null_pres` enum('V','N') COLLATE utf8_spanish_ci DEFAULT 'V' COMMENT 'las posiciones pueden anularse con basurero, bomba  o valor total=0\nsiempre es bueno tener respald de que posicion esta nula',
  `detalle` varchar(45) COLLATE utf8_spanish_ci DEFAULT NULL,
  `codigo_item_ps` varchar(10) COLLATE utf8_spanish_ci NOT NULL,
  `numero_pedido_ps` varchar(10) COLLATE utf8_spanish_ci NOT NULL,
  `id_cancelacion_ps` int(11) NOT NULL,
  PRIMARY KEY (`id`),
  KEY `fk_producto_solicitado_item1_idx` (`codigo_item_ps`),
  KEY `fk_producto_solicitado_pedido1_idx` (`numero_pedido_ps`),
  KEY `fk_producto_solicitado_cancelacion1_idx` (`id_cancelacion_ps`),
  CONSTRAINT `fk_producto_solicitado_cancelacion1` FOREIGN KEY (`id_cancelacion_ps`) REFERENCES `cancelacion` (`id`) ON DELETE NO ACTION ON UPDATE NO ACTION,
  CONSTRAINT `fk_producto_solicitado_item1` FOREIGN KEY (`codigo_item_ps`) REFERENCES `item` (`codigo`) ON DELETE NO ACTION ON UPDATE NO ACTION,
  CONSTRAINT `fk_producto_solicitado_pedido1` FOREIGN KEY (`numero_pedido_ps`) REFERENCES `pedido` (`numero`) ON DELETE NO ACTION ON UPDATE NO ACTION
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_spanish_ci
